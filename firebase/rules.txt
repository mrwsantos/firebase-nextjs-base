rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper function - verifica se usuário está autenticado
    function isAuthenticated() {
      return request.auth != null;
    }
    
    // Helper function - verifica se é o próprio usuário
    function isOwner(userId) {
      return isAuthenticated() && request.auth.uid == userId;
    }
    
    // Helper function - verifica se é admin ou master
    function isAdmin() {
      return isAuthenticated() && 
             get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role in ['admin', 'master'];
    }
    
    // Helper function - verifica se é master
    function isMaster() {
      return isAuthenticated() && 
             get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'master';
    }

    // ===== USERS COLLECTION =====
    match /users/{userId} {
      // Qualquer usuário autenticado pode ler seu próprio documento
      allow read: if isOwner(userId);
      
      // Admins e Masters podem ler qualquer usuário
      allow read: if isAdmin();
      
      // Usuário pode atualizar apenas alguns campos do próprio documento
      allow update: if isOwner(userId) && 
                      request.resource.data.diff(resource.data).affectedKeys()
                        .hasOnly(['name', 'updatedAt', 'photoURL']);
      
      // Apenas admins e masters podem criar novos usuários
      allow create: if isAdmin();
      
      // Apenas masters podem deletar usuários
      allow delete: if isMaster();
      
      // Admins podem atualizar qualquer campo de usuários
      allow update: if isAdmin();
    }

    // ===== COMPANIES COLLECTION (se você tiver) =====
    match /companies/{companyId} {
      // Usuários autenticados podem ler empresas que pertencem
      allow read: if isAuthenticated();
      
      // Apenas admins podem criar/atualizar/deletar empresas
      allow write: if isAdmin();
    }

    // ===== OUTRAS COLLECTIONS =====
    // Adicione regras específicas para outras collections conforme necessário
    
    // Bloqueia tudo que não foi explicitamente permitido
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
